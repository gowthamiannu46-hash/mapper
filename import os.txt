import os
import exifread
import pandas as pd
import folium


def get_decimal_from_dms(dms, ref):
    degrees = float(dms[0].num) / float(dms[0].den)
    minutes = float(dms[1].num) / float(dms[1].den)
    seconds = float(dms[2].num) / float(dms[2].den)

    decimal = degrees + minutes/60 + seconds/3600

    if ref in ['S','W']:
        decimal = -decimal

    return decimal


def process_images(folder):

    data = []

    for file in os.listdir(folder):

        if file.lower().endswith(".jpg"):

            path = os.path.join(folder, file)

            with open(path,'rb') as f:

                tags = exifread.process_file(f)

                if 'GPS GPSLatitude' in tags and 'GPS GPSLongitude' in tags:

                    lat = get_decimal_from_dms(
                        tags['GPS GPSLatitude'].values,
                        tags['GPS GPSLatitudeRef'].printable
                    )

                    lon = get_decimal_from_dms(
                        tags['GPS GPSLongitude'].values,
                        tags['GPS GPSLongitudeRef'].printable
                    )

                    data.append([file,lat,lon])

                else:
                    print("No GPS data:",file)

    df = pd.DataFrame(data,columns=["file","lat","lon"])
    return df


def create_map(df):

    m = folium.Map(
        location=[df.lat.mean(),df.lon.mean()],
        zoom_start=18
    )

    for i,row in df.iterrows():

        folium.Marker(
            [row.lat,row.lon],
            popup=row.file
        ).add_to(m)

    m.save("map.html")

    print("Map saved as map.html")


folder = r"D:\maping\images\footballcourt"

df = process_images(folder)

create_map(df)
